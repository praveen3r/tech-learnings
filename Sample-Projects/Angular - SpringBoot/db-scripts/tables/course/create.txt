CREATE TABLE COURSE
  (
    ID     NUMERIC ,
    NAME  VARCHAR(30) NOT NULL,
    TYPE  NUMERIC(2) NOT NULL,
    AUTHOR VARCHAR(100) NOT NULL,
    USER_ID NUMERIC,
    CRTD_DT TIMESTAMP NOT NULL,
    CONSTRAINT pk_course PRIMARY KEY (ID),
    CONSTRAINT fk_course_type FOREIGN KEY (TYPE) REFERENCES COURSE_TYPE_MASTER(ID),
    CONSTRAINT fk_user_id FOREIGN KEY (USER_ID) REFERENCES DEMO_USER(ID)
  );
