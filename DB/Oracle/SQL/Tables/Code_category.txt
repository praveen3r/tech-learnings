CREATE TABLE CODE_CATEGORY (CODE_CATEGORY_ID NUMBER(10) NOT NULL, CATEGORY_NAME NVARCHAR2(128) NOT NULL, CATEGORY_DESCRIPTION NVARCHAR2(128), IS_SYSTEM NVARCHAR2(1) NOT NULL, CONSTRAINT PK_CODE_CATEGORY PRIMARY KEY (CODE_CATEGORY_ID));
COMMENT ON TABLE CODE_CATEGORY IS 'This table contains category definitions for classifying [CODE] records.';
COMMENT ON COLUMN CODE_CATEGORY.CODE_CATEGORY_ID IS 'Server-generated identifier of the code category, based on the CODE_CATEGORYID sequence.';
COMMENT ON COLUMN CODE_CATEGORY.CATEGORY_NAME IS 'Logical name of the code category.';
COMMENT ON COLUMN CODE_CATEGORY.CATEGORY_DESCRIPTION IS 'Description of the category.';
COMMENT ON COLUMN CODE_CATEGORY.IS_SYSTEM IS 'Boolean flag (Y for yes or N for no) indicating whether the code category is for system codes, which users cannot edit.';

CREATE TABLE CODE_CATEGORY_ML (CODE_CATEGORY_ID NUMBER(10) NOT NULL, LANGUAGE_ID NVARCHAR2(3) NOT NULL, CATEGORY_DESCRIPTION NVARCHAR2(128), CONSTRAINT PK_CODE_CATEGORY_ML PRIMARY KEY (CODE_CATEGORY_ID, LANGUAGE_ID), CONSTRAINT FK02_CODE_CATEGORY_ML FOREIGN KEY (LANGUAGE_ID) REFERENCES "CORXPRADM"."LANGUAGE" ("LANGUAGE_ID"), CONSTRAINT FK01_CODE_CATEGORY_ML FOREIGN KEY (CODE_CATEGORY_ID) REFERENCES "CODE_CATEGORY" ("CODE_CATEGORY_ID"));
